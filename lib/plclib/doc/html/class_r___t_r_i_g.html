Detailed Description

Detect a rising edge.

Code sample: 

#include "plc_standardlib.h"
#define X0 2

R_TRIG trig;
void setup() {
    pinMode(X0, INPUT_PULLUP);
    Serial.begin(9600);
}
void loop() {
    boolean x0 = !digitalRead(X0);
    trig.process(x0);
    if (trig.Q) {
        Serial.println("This was a rising edge.");
    }
}
