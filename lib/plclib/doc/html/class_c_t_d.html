Detailed Description

Downward counter.

If CTD::LOAD is true the counter variable CTD::CV is initialised to the upper limit CTD::PV. If there is a rising edge on CTD::CD the counter value CTD::CV is decreased by one as long as CTD::CV is bigger than 0. The output variable CTD::Q is set to true if CTD::CV is equal to 0.

Code sample: #include "plc_standardlib.h" #define X0 2 #define X1 3

CTD ctd; R_TRIG rtrig;

void setup() { pinMode(X0, INPUT_PULLUP); pinMode(X1, INPUT_PULLUP); Serial.begin(9600); ctd.PV = 10; //set upper limit of counter }

void loop() { boolean x0 = !digitalRead(X0); //count up boolean x1 = !digitalRead(X1); //reset ctd.process(x0, x1); rtrig.process(x0 || x1); if (rtrig.Q) { Serial.print("Counter: "); Serial.print(ctd.CV); Serial.print(", "); Serial.print("Output: "); Serial.println(ctd.Q); } }
